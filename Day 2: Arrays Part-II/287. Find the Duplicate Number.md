/*
Brute : Sort the array and check for i and (i+1)th idx if both are same then that is ans
Better : Use hashing
Best : Use linklist cycle method. Use a fast and slow pointer and keep moving slow by 1 step 
       and fast by 2 steps till they collide. Then move fast to 1st element. Now again start              moving fast and slow by 1 step till they meet and point of collision is ans
*/

class Solution {
    public int findDuplicate(int[] nums) {
        int slow=nums[0],fast=nums[0];
        
        do{
            slow=nums[slow];
            fast=nums[nums[fast]];
        }while(slow!=fast);
        
        fast=nums[0];
        while(slow!=fast){
            slow=nums[slow];
            fast=nums[fast];
        }
        
        return slow;
    }
}
